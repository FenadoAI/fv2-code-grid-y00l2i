REAL-TIME LOGS

devforce.logs.stream('my-app', {
  onLog: (log) => console.log(log),
  filter: 'error'
});
for log in devforce.logs.stream('my-app'):
    if log.level == 'error':
        print(f"ERROR: {log.message}")
curl -N https://api.devforce.com/logs/my-app \
  -H "Accept: text/stream" \
  -H "Authorization: Bearer token"

AUTO SCALING

devforce.scale({
  min: 1,
  max: 100,
  cpu_threshold: 80
});
devforce.configure_scaling(
    min_instances=1,
    max_instances=100,
    cpu_threshold=80
)
curl -X PUT https://api.devforce.com/scale \
  -d '{"min":1,"max":100,"cpu":80}' \
  -H "Content-Type: application/json"
